def _extract_gstin(self) -> str:
    with pdfplumber.open(self.pdf_path) as pdf:
        page = pdf.pages[0]
        tables = page.extract_tables()

    for table in tables:
        for row in table:
            if row and "GSTIN of the supplier" in str(row[0]):
                return row[1].strip()

    raise ValueError("GSTIN not found in table")
